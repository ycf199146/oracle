oracle索引查询

COL INDEX_TYPE FOR A14
COL INDEX_OWNER FOR A15
COL COLUMN_NAME FOR A60
SELECT A.INDEX_OWNER,
       A.INDEX_NAME,
       LISTAGG(A.COLUMN_NAME, ',') WITHIN
GROUP(
ORDER BY A.INDEX_OWNER, A.INDEX_NAME, A.TABLE_NAME) COLUMN_NAME, B.INDEX_TYPE, B.UNIQUENESS
  FROM DBA_IND_COLUMNS A
INNER JOIN DBA_INDEXES B ON A.INDEX_OWNER = B.OWNER
                         AND A.TABLE_OWNER = B.TABLE_OWNER
                         AND A.TABLE_NAME = B.TABLE_NAME
                         AND A.INDEX_NAME = B.INDEX_NAME
WHERE A.TABLE_NAME = B.TABLE_NAME
   AND B.TABLE_NAME = 'EMR_KBM_BLLB'
   AND B.OWNER = 'BSSZSK_BANEW'
GROUP BY A.INDEX_OWNER,
          A.INDEX_NAME,
          A.TABLE_NAME,
          B.INDEX_TYPE,
          B.UNIQUENESS
ORDER BY 2;
